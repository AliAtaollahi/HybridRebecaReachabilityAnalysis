physicalclass HeaterWithSensor(3)
{
	knownrebecs{
		Controller controller;
	}
	statevars{
		@Real float tempr;
		@Real float timer;
	}

	HeaterWithSensor(float tempr_){
		tempr = tempr_;
		setMode(On);
	}

	mode On{
		inv(timer<=5){
			timer' = 1;
			tempr' = 1;
		}
		guard(timer>=5){
			timer = 0.0f;
			controller.control(tempr);
		}
	}

	mode Off{
		inv(timer<=5){
			timer' = 1;
			tempr' = -1;
		}
		guard(timer==5){
			timer = 0.0f;
			controller.control(tempr);
		}
	}
}

reactiveclass Controller(3)
{
	knownrebecs {
		HeaterWithSensor hws;
	}
	statevars {}

	msgsrv initial(){
	}

	msgsrv control(float tempr)
	{
		if(tempr >= 22)
			hws.setMode(Off);

		if(tempr <= 18)
			hws.setMode(On);
	}
}

main
{
	HeaterWithSensor hws(@Wire controller):(20.0f);
	Controller controller (@Wire hws):();
}
